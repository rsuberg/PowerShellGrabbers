#Ping Public IP Using Non-Onboard interface:
while ($true) {Clear-Host; Show-IP  -Table -Up; Get-NetAdapter | where MediaConnectionState -eq "Connected" | select-object ifIndex, Name, InterfaceDescription, MacAddress, LinkSpeed, MediaType, AdminStatus, MediaConnectionState, PromiscuousMode | where AdminStatus -eq 'Up' | where MediaType -like "802*" | sort ifindex | ft -AutoSize;(Get-NetRoute "0.0.0.0/0") | sort ifindex | ft ifIndex, InterfaceAlias, InterfaceMetric, RouteMetric, NextHop, DestinationPrefix; ping -S ( Get-NetIPAddress -AddressFamily IPv4 -ifIndex (Get-NetAdapter | where status -eq "up" | where ifIndex -ne 5).ifIndex).IPAddress -n 1 1.1.1.1 | findstr "Destination timed Reply";sleep 3}

#Ping with FPING to limit to reply.
while ($true) {Clear-Host; Show-IP  -Table -Up; Get-NetAdapter | where MediaConnectionState -eq "Connected" | select-object ifIndex, Name, InterfaceDescription, MacAddress, LinkSpeed, MediaType, AdminStatus, MediaConnectionState, PromiscuousMode | where AdminStatus -eq 'Up' | where MediaType -like "802*" | sort ifindex | ft -AutoSize;(Get-NetRoute "0.0.0.0/0") | sort ifindex | ft ifIndex, InterfaceAlias, InterfaceMetric, RouteMetric, NextHop, DestinationPrefix; c:\Tools\Fping.exe 1.1.1.1 -l -n 1 ;sleep 3}
